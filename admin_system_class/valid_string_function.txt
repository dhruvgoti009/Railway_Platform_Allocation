bool admin_system :: check_valid_string(string str)
{
    if(str.size()!=5)
        return 0;
    if(str[2]!=':')
        return 0;
    for(int i=0;i<5;i++)
    {
        if(i!=2)
        {
            if(!(str[i]>='0' && str[i]<='9'))
                return 0;
        }
    }
    return 1;
}
